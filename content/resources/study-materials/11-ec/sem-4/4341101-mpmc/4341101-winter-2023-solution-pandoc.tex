\begin{verbatim}

+{-{-}{-}{-}{-}{-}{-}{-}+{-}{-}{-}{-}{-}{-}{-}{-}+{-}{-}{-}{-}{-}{-}{-}{-}+}

{\def\LTcaptype{none} % do not increment counter

\endhead
\bottomrule\noalign{}

\item
  \textbf{ALU}: Handles ADD/SUB/AND/OR/XOR operations on data
\end{itemize}

|  +{-{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}+  +{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}+  |}
|        |               |         |
|        v               v         |
|  +{-{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}+  +{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}+  |}
|  |   ALU      |{{-}|Registers   |  |}
|  |            |{-|            |  |}
|  +{-{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}+  +{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}+  |}
+{-{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}{-}+}

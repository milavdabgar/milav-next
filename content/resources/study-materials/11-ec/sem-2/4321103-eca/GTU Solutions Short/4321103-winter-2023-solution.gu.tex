\questionmarks{1}{a}{3}
\textbf{ટ્રાન્ઝિસ્ટર બાયસિંગ શું છે? તેની શું જરૂર છે?}

\begin{solutionbox}
\textbf{જવાબ}:
ટ્રાન્ઝિસ્ટર બાયસિંગ એ AC સિગ્નલના યોગ્ય એમ્પ્લિફિકેશન માટે સ્થિર DC ઓપરેટિંગ પોઈન્ટ (Q-પોઈન્ટ) સ્થાપિત કરવાની પ્રક્રિયા છે.

\begin{center}
\captionof{table}{ટ્રાન્ઝિસ્ટર બાયસિંગની જરૂરિયાત}
\begin{tabular}{|l|l|}
\hline
\textbf{પાસું} & \textbf{મહત્વ} \\ \hline
સ્થિરતા & તાપમાન વધઘટ છતાં સ્થિર Q-પોઈન્ટ જાળવે છે \\ \hline
લિનિયરતા & વિકૃતિ-મુક્ત એમ્પ્લિફિકેશન માટે લિનિયર રીજનમાં કાર્ય સુનિશ્ચિત કરે છે \\ \hline
કાર્યક્ષમતા & સિગ્નલ ક્લિપિંગ અટકાવે છે અને સિગ્નલ સ્વિંગને મહત્તમ કરે છે \\ \hline
વિશ્વસનીયતા & થર્મલ રનઅવે ટાળે છે અને ટ્રાન્ઝિસ્ટરને સુરક્ષિત રાખે છે \\ \hline
\end{tabular}
\end{center}
\end{solutionbox}
\begin{mnemonicbox}
``SOLE ઓપરેશન'' (Stability, Operating point, Linearity, Efficiency)
\end{mnemonicbox}

\questionmarks{1}{b}{4}
\textbf{CE એમ્પ્લિફાયર માટે લોડ લાઇન સમજાવો}

\begin{solutionbox}
\textbf{જવાબ}:
લોડ લાઇન એ ટ્રાન્ઝિસ્ટર સર્કિટના બધા સંભવિત ઓપરેટિંગ પોઈન્ટનું ગ્રાફિકલ રેપ્રેઝન્ટેશન છે.

\begin{center}
\includefigure{figures/tex-diagrams/pdf/4321103-winter-2023-solution-q1b.gu.pdf}
\end{center}

\begin{itemize}
    \item \textbf{DC લોડ લાઇન}: સેચુરેશન પોઈન્ટ ($Ic=Vcc/Rc, Vce=0$) અને કટઓફ પોઈન્ટ ($Ic=0, Vce=Vcc$) વચ્ચે દોરાય છે
    \item \textbf{AC લોડ લાઇન}: Q-પોઈન્ટમાંથી પસાર થાય છે, સ્લોપ = $-1/r_c$ ($r_c$ = AC કલેક્ટર રેસિસ્ટન્સ)
    \item \textbf{Q-પોઈન્ટ}: ઓપરેટિંગ પોઈન્ટ જ્યાં DC બાયસિંગ કન્ડિશન્સ સ્થાપિત થાય છે
\end{itemize}
\end{solutionbox}
\begin{mnemonicbox}
``SCQ પોઈન્ટ્સ'' (Saturation, Cutoff, Q-point)
\end{mnemonicbox}

\questionmarks{1}{c}{7}
\textbf{ટ્રાન્ઝિસ્ટરની વિવિધ બાયસિંગ પધ્ધતિની યાદી બનાવો અને તેમાથી કોઈપણ એક સમજાવો.}

\begin{solutionbox}
\textbf{જવાબ}:
ટ્રાન્ઝિસ્ટર માટેની વિવિધ બાયસિંગ પધ્ધતિઓ:

\begin{center}
\captionof{table}{ટ્રાન્ઝિસ્ટર બાયસિંગ પધ્ધતિઓ}
\begin{tabular}{|l|l|}
\hline
\textbf{પધ્ધતિ} & \textbf{મુખ્ય લક્ષણ} \\ \hline
ફિક્સ્ડ બાયસ & બેઝ બાયસ માટે એક રેસિસ્ટર \\ \hline
કલેક્ટર-ટુ-બેઝ બાયસ & નેગેટિવ ફીડબેક દ્વારા સેલ્ફ-સ્ટેબિલાઈઝિંગ \\ \hline
વોલ્ટેજ ડિવાઈડર બાયસ & વોલ્ટેજ ડિવાઈડર નેટવર્ક દ્વારા સૌથી સ્થિર \\ \hline
એમિટર બાયસ & એમિટર રેસિસ્ટર સાથે ઉત્તમ સ્થિરતા \\ \hline
કોમ્બિનેશન બાયસ & ઓપ્ટિમલ સ્થિરતા માટે મલ્ટિપલ ફીડબેક પાથનો ઉપયોગ \\ \hline
\end{tabular}
\end{center}

\textbf{વોલ્ટેજ ડિવાઈડર બાયસ સમજૂતી:}

\begin{center}
\includefigure{figures/tex-diagrams/pdf/4321103-winter-2023-solution-q1c.gu.pdf}
\end{center}

\begin{itemize}
    \item \textbf{ઓપરેશન}: $R_1$ અને $R_2$ બેઝ વોલ્ટેજ સેટ કરવા માટે વોલ્ટેજ ડિવાઈડર બનાવે છે
    \item \textbf{સ્થિરતા}: સ્ટિફ વોલ્ટેજ ડિવાઈડરને કારણે ઉત્તમ થર્મલ સ્થિરતા
    \item \textbf{કાર્યક્ષમતા}: $\beta$ વેરિએશનથી સ્વતંત્ર હોવાથી સૌથી વધુ ઉપયોગમાં લેવાતી પધ્ધતિ
    \item \textbf{ગણતરી}: બેઝ વોલ્ટેજ = $V_{CC} \times R_2/(R_1+R_2)$
\end{itemize}
\end{solutionbox}
\begin{mnemonicbox}
``VISE ગ્રિપ'' (Voltage divider, Independent of $\beta$, Stable, Efficient)
\end{mnemonicbox}

\orquestionmarks{1}{c}{7}
\textbf{સર્કિટ ડાયગ્રામની મદદથી વોલ્ટેજ ડિવાઈડર બાયસિંગ પધ્ધતિ સમજાવો}

\begin{solutionbox}
\textbf{જવાબ}:
વોલ્ટેજ ડિવાઈડર બાયસિંગ એ ટ્રાન્ઝિસ્ટરને બાયસ કરવાની સૌથી સ્થિર પદ્ધતિ છે.

\begin{center}
\includefigure{figures/tex-diagrams/pdf/4321103-winter-2023-solution-q1c-2.gu.pdf}
\end{center}

\begin{center}
\captionof{table}{વોલ્ટેજ ડિવાઈડર બાયસિંગની વિશેષતાઓ}
\begin{tabular}{|l|l|}
\hline
\textbf{કોમ્પોનન્ટ} & \textbf{કાર્ય} \\ \hline
$R_1, R_2$ & $\beta$ થી સ્વતંત્ર સ્થિર બેઝ વોલ્ટેજ બનાવે છે \\ \hline
$R_C$ & કલેક્ટર કરંટને મર્યાદિત કરે છે અને આઉટપુટ વોલ્ટેજ વિકસિત કરે છે \\ \hline
$R_E$ & નેગેટિવ ફીડબેક દ્વારા સ્થિરતા પ્રદાન કરે છે \\ \hline
બાયપાસ કેપેસિટર & ગેઇન વધારવા માટે $R_E$ ની આસપાસ AC સિગ્નલને બાયપાસ કરે છે \\ \hline
\end{tabular}
\end{center}

\begin{itemize}
    \item \textbf{કાર્યરત સિદ્ધાંત}: $R_1$ અને $R_2$ બેઝ વોલ્ટેજ સેટ કરતા વોલ્ટેજ ડિવાઈડર બનાવે છે
    \item \textbf{થર્મલ સ્થિરતા}: $R_E$ નેગેટિવ ફીડબેક માટે ઉત્તમ થર્મલ સ્થિરતા પ્રદાન કરે છે
    \item \textbf{ફાયદો}: તાપમાન અને $\beta$ માં ફેરફાર છતાં Q-પોઈન્ટ સ્થિર રહે છે
\end{itemize}
\end{solutionbox}
\begin{mnemonicbox}
``BEST બાયસ'' (Base voltage, Emitter stability, Stiff divider, Temperature stable)
\end{mnemonicbox}

\orquestionmarks{2}{b}{4}
\textbf{સિંગલ સ્ટેજ એમ્પ્લિફાયરનો ફ્રિક્વન્સી રિસ્પોન્સ સમજાવો અને તેની કટઓફ ફ્રિક્વન્સીઓ દર્શાવો.}

ઓછી ગરમી ઉત્પન્ન કરે છે & મુશ્કેલ ટ્રબલશૂટિંગ \\ \hline

    \item \textbf{ઉપયોગો}: કમ્પ્યુટર, ટીવી, મોબાઈલ ચાર્જર, LED ડ્રાઈવર

\hline

વોલ્ટેજની વધઘટ સામે રક્ષણ આપે છે & નિયમિત બેટરી મેન્ટેનન્સ \\ \hline

સ્મૂધ પાવર ટ્રાન્ઝિશન & ઓપરેશન દરમિયાન નોઈઝ \\ \hline
પાવર કન્ડિશનિંગ & સ્ટેન્ડબાયમાં ઓછી કાર્યક્ષમતા \\ \hline
\end{tabular}
\end{center}

\begin{itemize}
    \item \textbf{પ્રકારો}: ઓફલાઇન/સ્ટેન્ડબાય, લાઇન-ઇન્ટરેક્ટિવ, ઓનલાઇન/ડબલ-કન્વર્ઝન
    \item \textbf{ઉપયોગો}: કમ્પ્યુટર, મેડિકલ ઇક્વિપમેન્ટ, ડેટા સેન્ટર, ટેલિકોમ્યુનિકેશન
    \item \textbf{કાર્ય}: સામાન્ય રીતે મેઈન પાવર પસાર કરે છે જ્યારે બેટરી ચાર્જ કરે છે; બ્લેકઆઉટ દરમિયાન બેટરી પાવર પર સ્વિચ કરે છે
\end{itemize}
\end{solutionbox}
\begin{mnemonicbox}
``POWER બેકઅપ'' (Protection from Outages, Waveform conditioning, Emission-free, Reliability boost)
\end{mnemonicbox}

    \item \keyword{FFT}: Separates orthogonal subcarriers.
    \item \keyword{Decoder}: Corrects errors and recovers data.
\end{itemize}
\end{solutionbox}

\begin{mnemonicbox}
\mnemonic{Receive Convert Remove Transform Decode}
\end{mnemonicbox}

    printf("Enter second floating point number: ");

    // Find the ceil of smaller number and floor of larger number
    if(num1 {} num2) \{
        start = ceil(num1);
        end = floor(num2);

    \}

    return 0;
\}

\textbf{Explain the working of switch-case statement with example.}

\begin{solutionbox}

The switch-case statement is a multi-way decision maker that tests the
value of an expression against various case values and executes the
matching case block.

\begin{verbatim}
\#include {stdio.h}

int main() \{
    int day;
    
    printf("Enter day number (1{-7): "});
    scanf("\%d", \&day);
    
    switch(day) \{
        case 1:
            printf("Monday{n}");
            break;
        case 2:
            printf("Tuesday{n}");
            break;
        case 3:
            printf("Wednesday{n}");
            break;
        case 4:
            printf("Thursday{n}");
            break;
        case 5:
            printf("Friday{n}");
            break;
        case 6:
            printf("Saturday{n}");
            break;
        case 7:
            printf("Sunday{n}");
            break;
        default:
            printf("Invalid day number{n}");
    \}
    
    return 0;
\}
\end{verbatim}

\textbf{Diagram:}

\begin{verbatim}
flowchart LR
    A[/Input day/] {-{-} B\{"switch(day)"\}}
    B {-{-} C1\{case 1\}}
    B {-{-} C2\{case 2\}}
    B {-{-} C3\{...\}}

\begin{solutionbox}

\begin{verbatim}
\#include {stdio.h}
\#include {math.h}

int main() \{
    float number;
    int result;

    scanf("\%f", \&number);

\textbf{Program with Square Function:}

\begin{verbatim}
\#include {stdio.h}

// Function to calculate square of an integer
int square(int num) \{

    int number, result;
    
    // Input an integer
    printf("Enter an integer: ");
    scanf("\%d", \&number);
    
    // Call the square function
    result = square(number);
    
    // Display the result
    printf("Square of \%d is \%d{n}", number, result);
    
    return 0;
\}
\end{verbatim}

    \item \textbf{ટર્ન-ઓફ}: એનોડ કરંટ હોલ્ડિંગ કરંટથી નીચે જવો જોઈએ
\end{itemize}
\end{solutionbox}
\mnemonicbox{PPFF - Positive feedback Perpetuates Forward conduction}

    \item \textbf{ઝીરો-ક્રોસિંગ}: માત્ર ઝીરો વોલ્ટેજ પર સ્વિચિંગ EMI/RFI નોઇઝ ઘટાડે છે
    \item \textbf{સાયલેન્ટ ઓપરેશન}: પરંપરાગત રિલેથી વિપરીત, કોઈ મેકેનિકલ ક્લિક નથી
    \item \textbf{લાંબી લાઇફ}: પરંપરાગત રિલેમાં જેવા મેકેનિકલ ઘસારો નથી
\end{itemize}
\end{solutionbox}

\textbf{DC મોટર્સમાં હોલ ઇફેક્ટ સેન્સર:}

\begin{tabulary}{\linewidth}{L L}
    \hline
    \textbf{પાસું} & \textbf{વર્ણન} \\
    \hline
    વ્યાખ્યા & એગ્નેટિક ફિલ્ડને ડિટેક્ટ કરતા સેમિકન્ડક્ટર-આધારિત સેન્સર \\
    સિદ્ધાંત & મેગ્નેટિક ફિલ્ડમાં કરંટ ફ્લોથી લંબરૂપે વોલ્ટેજ ડિફરન્સ ઉત્પન્ન થાય છે \\
    સિગ્નલ આઉટપુટ & ડિજિટલ (ON/OFF) અથવા એનાલોગ (ફિલ્ડ સ્ટ્રેન્થના પ્રમાણમાં) \\
    સાઇઝ & કોમ્પેક્ટ, મોટર હાઉસિંગમાં ઇન્ટિગ્રેટેડ થઈ શકે છે \\
    \hline
\end{tabulary}

\textbf{DC મોટર્સમાં રોલ:}

\begin{tabulary}{\linewidth}{L L L}
    \hline
    \textbf{ફંક્શન} & \textbf{એપ્લિકેશન} & \textbf{બેનિફિટ} \\
    \hline
    પોઝિશન સેન્સિંગ & રોટર પોઝિશન ડિટેક્શન & પ્રિસાઇઝ કોમ્યુટેશન ટાઇમિંગ \\
    સ્પીડ મેઝરમેન્ટ & RPM કેલ્ક્યુલેશન માટે પલ્સ જનરેશન & એક્યુરેટ સ્પીડ ફીડબેક \\
    ડિરેક્શન ડિટેક્શન & ફેઝ સિક્વન્સ મોનિટરિંગ & રોટેશન ડિરેક્શન કંટ્રોલ \\
    કરંટ સેન્સિંગ & નોન-કોન્ટેક્ટ કરંટ મેઝરમેન્ટ & ઓવરલોડ પ્રોટેક્શન \\
    \hline
\end{tabulary}

\begin{itemize}
    \item \textbf{BLDC મોટર્સ}: ઇલેક્ટ્રોનિક કોમ્યુટેશન (મિકેનિકલ કોમ્યુટેટરને રિપ્લેસ કરવા) માટે ક્રિટિકલ
    \item \textbf{પ્રિસિઝન}: મિકેનિકલ સેન્સર કરતાં ઉચ્ચ ચોકસાઈ
    \item \textbf{રિલાયબિલિટી}: કોઈ મિકેનિકલ ઘસારો નહીં, લાંબી સર્વિસ લાઇફ
    \item \textbf{ઇન્ટિગ્રેશન}: ડ્રાઇવ ઇલેક્ટ્રોનિક્સ સાથે ઇન્ટિગ્રેટેડ થઈ શકે છે
\end{itemize}
\end{solutionbox}

    \item \textbf{ઓપન-લૂપ કંટ્રોલ}: સામાન્ય રીતે પોઝિશન ફીડબેકની જરૂર નથી

    CPU/પ્રોસેસર & પ્રોગ્રામ એક્ઝિક્યુટ કરે છે, ઓપરેશન્સ નિયંત્રિત કરે છે & સ્પીડ સ્કેન ટાઇમમાં માપવામાં આવે છે (ms) \\
    ઇનપુટ ઇન્ટરફેસ & સેન્સર અને સ્વિચ સાથે કનેક્ટ કરે છે & ડિજિટલ/એનાલોગ, આઇસોલેશન, ફિલ્ટરિંગ \\
    આઉટપુટ ઇન્ટરફેસ & એક્ચુએટર અને ઇન્ડિકેટર સાથે કનેક્ટ કરે છે & રિલે/ટ્રાન્ઝિસ્ટર/ટ્રાયક આઉટપુટ \\
    મેમરી & પ્રોગ્રામ અને ડેટા સ્ટોર કરે છે & પ્રોગ્રામ, ડેટા, અને સિસ્ટમ મેમરી એરિયા \\
    પ્રોગ્રામિંગ ડિવાઇસ & પ્રોગ્રામ્સ ડેવલપ અને લોડ કરવા માટે વપરાય છે & PC, હેન્ડહેલ્ડ પ્રોગ્રામર, સોફ્ટવેર \\
    કમ્યુનિકેશન & નેટવર્ક/અન્ય ડિવાઇસિસ સાથે કનેક્ટ કરે છે & ઔદ્યોગિક પ્રોટોકોલ, રિમોટ I/O \\
    \hline
\end{tabulary}

\begin{itemize}
    \item \textbf{સ્કેન સાયકલ}: ઇનપુટ વાંચવા, પ્રોગ્રામ એક્ઝિક્યુટ કરવા, આઉટપુટ અપડેટ કરવાની ક્રમિક પ્રક્રિયા
    \item \textbf{પ્રોગ્રામિંગ લેંગ્વેજિસ}: લેડર ડાયાગ્રામ (LD), ફંક્શન બ્લોક ડાયાગ્રામ (FBD), સ્ટ્રક્ચર્ડ ટેક્સ્ટ (ST), ઇન્સ્ટ્રક્શન લિસ્ટ (IL), સિક્વેન્શિયલ ફંક્શન ચાર્ટ (SFC)
    \item \textbf{મોડ્યુલરિટી}: વધારાના I/O મોડ્યુલ્સ સાથે વિસ્તૃત કરી શકાય છે
    \item \textbf{રોબસ્ટનેસ}: કઠોર ઔદ્યોગિક પર્યાવરણ માટે ડિઝાઇન કરેલ
    \item \textbf{રિલાયાબિલિટી}: સામાન્ય રીતે MTBF $>100,000$ કલાક
\end{itemize}
\end{solutionbox}

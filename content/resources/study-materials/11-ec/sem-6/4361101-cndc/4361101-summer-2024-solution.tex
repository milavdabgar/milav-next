\begin{itemize}
    \item \textbf{Stop-and-Wait}: Send one frame, wait for ACK
    \item \textbf{Sliding Window}: Send multiple frames before ACK
    \item \textbf{Buffer Management}: Prevent receiver overflow
\end{itemize}

\begin{itemize}

\end{itemize}

\begin{itemize}
\tightlist
\item
  \textbf{Collision Domain}: Does not segment collisions
\item
  \textbf{No Filtering}: Forwards all signals
\end{itemize}

\textbf{Hub Characteristics:}

\begin{itemize}
\tightlist

  \textbf{Shared Medium}: All ports share bandwidth
\item

\item

\end{itemize}

\textbf{Switch Characteristics:}

\begin{itemize}
\tightlist
\item
  \textbf{MAC Address Table}: Learns device locations
\item
  \textbf{Full Duplex}: Simultaneous send/receive
\item
  \textbf{VLAN Support}: Virtual network segmentation
\end{itemize}

\textbf{Router Characteristics:}

\begin{itemize}
\tightlist
\item
  \textbf{IP Routing}: Forwards packets between networks
\item
  \textbf{Routing Table}: Maintains network topology
\item
  \textbf{NAT Support}: Network Address Translation
\end{itemize}

\textbf{Applications:}

\begin{itemize}
\tightlist
\item
  \textbf{Hub}: Legacy networks (mostly obsolete)
\item
  \textbf{Switch}: LAN connectivity, VLAN implementation
\item
  \textbf{Router}: Internet connectivity, WAN connections
\end{itemize}

\end{solutionbox}

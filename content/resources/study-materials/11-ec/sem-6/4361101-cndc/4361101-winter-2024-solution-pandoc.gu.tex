\subsection*{પ્રશ્ન 1(અ) [3
ગુણ]}\label{uxaaauxab0uxab6uxaa8-1uxa85-3-uxa97uxaa3}

\textbf{સ્ટાર ટોપોલોજીનું સવિસ્તાર વર્ણન કરો.}

\begin{solutionbox}
સ્ટાર ટોપોલોજીમાં બધા devices એક કેન્દ્રીય hub અથવા switch સાથે
જોડાયેલા હોય છે. દરેક device નો કેન્દ્રીય device સાથે અલગ point-to-point
connection હોય છે.

\textbf{આકૃતિ:}

\begin{verbatim}
    Computer A
        |
        |
Computer D {-{-}{-}{-} HUB {-}{-}{-}{-} Computer B}
        |
        |
    Computer C
\end{verbatim}

\textbf{મુખ્ય લક્ષણો:}

\begin{itemize}
\tightlist
\item
  \textbf{કેન્દ્રીય હબ}: બધા connections કેન્દ્રીય device મારફતે પસાર થાય છે
\item
  \textbf{સમર્પિત લિંક્સ}: દરેક node નો અલગ connection હોય છે
\item
  \textbf{સરળ મેનેજમેન્ટ}: devices ને add/remove કરવું સરળ હોય છે
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``સ્ટાર કેન્દ્રમાં ચમકે'' - બધા devices કેન્દ્રીય બિંદુ સાથે
જોડાય છે

\end{mnemonicbox}
\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsection*{પ્રશ્ન 1(બ) [4

servers પાસેથી services માંગે છે. સર્વર અનેક clients ને resources અને services
પ્રદાન કરે છે.

\subsection*{પ્રશ્ન 1(ક) [7
ગુણ]}\label{uxaaauxab0uxab6uxaa8-1uxa95-7-uxa97uxaa3}

\textbf{TCP/IP મોડેલના દરેક લેયરના કાર્ય સાથે વર્ણન કરો.}

\begin{solutionbox}
TCP/IP મોડેલમાં ચાર layers છે જે networks પર end-to-end
communication પ્રદાન કરે છે.


{\def\LTcaptype{none} % do not increment counter
\vspace{-5pt}
\captionof{table}{TCP/IP મોડેલ લેયર્સ}
\vspace{-10pt}
\begin{longtable}[]{@{}lll@{}}
\toprule\noalign{}
લેયર & કાર્ય & પ્રોટોકોલ્સ \\
\midrule\noalign{}
\endhead
\bottomrule\noalign{}
\endlastfoot
Application & યુઝર interface, network services & HTTP, FTP, SMTP \\
Transport & End-to-end delivery, error control & TCP, UDP \\
Internet & Routing, logical addressing & IP, ICMP, ARP \\
Network Access & Physical transmission & Ethernet, WiFi \\
\end{longtable}
}

\textbf{લેયર કાર્યો:}

\begin{itemize}
\tightlist
\item
  \textbf{Application Layer}: યુઝર applications ને network services પ્રદાન
  કરે છે
\item
  \textbf{Transport Layer}: processes વચ્ચે વિશ્વસનીય data delivery સુનિશ્ચિત
  કરે છે
\item
  \textbf{Internet Layer}: IP વાપરીને multiple networks પર packets route
  કરે છે
\item
  \textbf{Network Access Layer}: data નું physical transmission હેન્ડલ કરે છે
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``બધા પરિવહન ઇન્ટરનેટ નેટવર્ક'' (ATIN)

\end{mnemonicbox}
\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsection*{પ્રશ્ન 1(ક OR) [7
ગુણ]}\label{uxaaauxab0uxab6uxaa8-1uxa95-or-7-uxa97uxaa3}

\textbf{OSI રેફરન્સ મોડેલના ડેટા લિંક લેયર અને નેટવર્ક લેયરની વિશેષતાઓ વર્ણવો.}

\begin{solutionbox}
ડેટા લિંક અને નેટવર્ક લેયર્સ OSI મોડેલમાં વિશ્વસનીય transmission અને
routing capabilities પ્રદાન કરે છે.


{\def\LTcaptype{none} % do not increment counter
\vspace{-5pt}
\captionof{table}{લેયર તુલના}
\vspace{-10pt}
\begin{longtable}[]{@{}lll@{}}
\toprule\noalign{}
લક્ષણ & ડેટા લિંક લેયર & નેટવર્ક લેયર \\
\midrule\noalign{}
\endhead
\bottomrule\noalign{}
\endlastfoot
મુખ્ય કાર્ય & Node-to-node delivery & End-to-end delivery \\
Addressing & MAC addresses & IP addresses \\
Error Control & Frame-level & Packet-level \\
\end{longtable}
}

\textbf{ડેટા લિંક લેયર કાર્યો:}

\begin{itemize}
\tightlist
\item
  \textbf{Framing}: bits ને frames માં વ્યવસ્થિત કરે છે
\item
  \textbf{Error Control}: transmission errors શોધે અને સુધારે છે
\item
  \textbf{Flow Control}: data transmission rate મેનેજ કરે છે
\end{itemize}

\textbf{નેટવર્ક લેયર કાર્યો:}

\begin{itemize}
\tightlist
\item
  \textbf{Routing}: packets માટે શ્રેષ્ઠ path નક્કી કરે છે
\item
  \textbf{Logical Addressing}: identification માટે IP addresses વાપરે છે
\item
  \textbf{Packet Forwarding}: networks વચ્ચે packets route કરે છે
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``ડેટા સ્થાનિક રીતે લિંક કરે, નેટવર્ક વૈશ્વિક રીતે route કરે''

\end{mnemonicbox}

\subsection*{પ્રશ્ન 2(ક) [7
ગુણ]}\label{uxaaauxab0uxab6uxaa8-2uxa95-7-uxa97uxaa3}

\textbf{FDDI અને CDDI નું વર્ણન કરો.}

\begin{solutionbox}
FDDI અને CDDI ring-based network technologies છે જે
high-speed data transmission પ્રદાન કરે છે.


{\def\LTcaptype{none} % do not increment counter
\vspace{-5pt}
\captionof{table}{FDDI vs CDDI તુલના}
\vspace{-10pt}
\begin{longtable}[]{@{}lll@{}}
\toprule\noalign{}
લક્ષણ & FDDI & CDDI \\
\midrule\noalign{}
\endhead
\bottomrule\noalign{}
\endlastfoot
માધ્યમ & Fiber optic & Copper (UTP) \\
ઝડપ & 100 Mbps & 100 Mbps \\
અંતર & 200 km & 100 meters \\
ખર્ચ & વધુ & ઓછો \\
\end{longtable}
}

\textbf{FDDI લક્ષણો:}

\begin{itemize}
\tightlist
\item
  \textbf{દ્વિ રિંગ}: fault tolerance માટે primary અને secondary rings
\item
  \textbf{ટોકન પાસિંગ}: deterministic access method
\item
  \textbf{સ્વ-નિકાલ}: failures પાસેથી automatic recovery
\end{itemize}

\textbf{CDDI લક્ષણો:}

\begin{itemize}
\tightlist
\item
  \textbf{કોપર માધ્યમ}: unshielded twisted pair cables વાપરે છે
\item
  \textbf{સમાન પ્રોટોકોલ}: transmission medium સિવાય FDDI જેવું જ
\item
  \textbf{કિંમત અસરકારક}: FDDI કરતાં ઓછી implementation cost
\end{itemize}

\textbf{રિંગ સ્ટ્રક્ચર:}

\begin{verbatim}
    Station A
        |
Station D {-{-}+{-}{-} Station B}
        |
    Station C
\end{verbatim}

\end{solutionbox}
\begin{mnemonicbox}
``FDDI ફાઇબર ઝડપી, CDDI કોપર સસ્તું''

\end{mnemonicbox}

\subsection*{પ્રશ્ન 2(અ OR) [3

\textbf{ફાયરવોલ ડેટાને કેવી રીતે સુરક્ષિત કરે છે.}

\begin{solutionbox}
ફાયરવોલ વિશ્વસનીય આંતરિક network અને અવિશ્વસનીય બાહ્ય networks
વચ્ચે security barrier તરીકે કામ કરે છે.

\textbf{સુરક્ષા પદ્ધતિઓ:}

\begin{itemize}
\tightlist
\item
  \textbf{પેકેટ ફિલ્ટરિંગ}: security rules માટે packet headers તપાસે છે
\item
  \textbf{એક્સેસ કંટ્રોલ}: અનધિકૃત access attempts ને block કરે છે
\item
  \textbf{ટ્રાફિક મોનિટરિંગ}: બધા incoming અને outgoing traffic ની દેખરેખ કરે
  છે
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``ફાયરવોલ દુશ્મનોને ફિલ્ટર કરે''

\end{mnemonicbox}
\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsection*{પ્રશ્ન 2(બ OR) [4
ગુણ]}\label{uxaaauxab0uxab6uxaa8-2uxaac-or-4-uxa97uxaa3}

\textbf{FDDI નું structure સમજાવો અને તેના ફાયદાઓ જણાવો.}

\begin{solutionbox}
FDDI high-speed, fault-tolerant networking માટે dual
counter-rotating rings વાપરે છે.

\textbf{સ્ટ્રક્ચર ઘટકો:}

\begin{itemize}
\tightlist
\item
  \textbf{પ્રાઇમરી રિંગ}: મુખ્ય data transmission path
\item
  \textbf{સેકન્ડરી રિંગ}: fault recovery માટે backup path
\item
  \textbf{ડ્યુઅલ એટેચમેન્ટ સ્ટેશન્સ}: બંને rings સાથે જોડાય છે
\item
  \textbf{સિંગલ એટેચમેન્ટ સ્ટેશન્સ}: એક ring સાથે જ જોડાય છે
\end{itemize}

\textbf{ફાયદાઓ:}

\begin{itemize}
\tightlist
\item
  \textbf{હાઇ સ્પીડ}: 100 Mbps transmission rate
\item
  \textbf{ફોલ્ટ ટોલરન્સ}: secondary ring વાપરીને automatic recovery
\item
  \textbf{લાંબુ અંતર}: 200 km સુધીના networks સાપોર્ટ કરે છે
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``FDDI દ્વિ રિંગ વિશ્વસનીયતા આપે''

\end{mnemonicbox}

  \textbf{બેસ્ટ એફર્ટ}: delivery અથવા order ની કોઈ ગેરંટી નથી
\item
  \textbf{લો ઓવરહેડ}: ઝડપી transmission માટે મિનિમલ હેડર
\end{itemize}

\end{solutionbox}

\textbf{ઉદાહરણ સાથે VoIP નું વર્ણન કરો.}

\begin{solutionbox}
વૉઇસ ઓવર ઇન્ટરનેટ પ્રોટોકોલ પરંપરાગત ટેલિફોન સિસ્ટમ્સ બદલે IP
networks પર voice communications ટ્રાન્સમિટ કરે છે.

\textbf{VoIP ઘટકો:}

\begin{itemize}
\tightlist
\item
  \textbf{IP ફોન}: VoIP કૉલ્સ માટે હાર્ડવેર device
\item
  \textbf{સોફ્ટફોન}: કમ્પ્યુટર-બેસ્ડ કૉલ્સ માટે સોફ્ટવેર એપ્લિકેશન
\item
  \textbf{ગેટવે}: VoIP ને પરંપરાગત phone networks સાથે જોડે છે
\item
  \textbf{PBX}: બિઝનેસ phone systems માટે પ્રાઇવેટ બ્રાન્ચ એક્સચેન્જ
\end{itemize}

\textbf{VoIP પ્રક્રિયા:}

\begin{enumerate}
\tightlist
\item
  \textbf{વૉઇસ કેપ્ચર}: માઇક્રોફોન voice ને analog signal માં convert કરે છે
\item
  \textbf{ડિજિટાઇઝેશન}: ADC analog ને digital samples માં convert કરે છે
\item
  \textbf{કમ્પ્રેશન}: કોડેક audio data ને compress કરે છે
\item
  \textbf{પેકેટાઇઝેશન}: voice data ને IP packets માં વિભાજિત કરે છે
\item
  \textbf{ટ્રાન્સમિશન}: packets IP network પર મોકલવામાં આવે છે
\item
  \textbf{પુનર્નિર્માણ}: receiving end audio ને reassemble અને play કરે છે
\end{enumerate}

\textbf{ઉદાહરણ એપ્લિકેશન્સ:}

\begin{itemize}
\tightlist
\item
  \textbf{સ્કાઇપ}: વ્યક્તિગત કૉલ્સ માટે કન્ઝ્યુમર VoIP સર્વિસ
\item
  \textbf{વોટ્સએપ કૉલિંગ}: મોબાઇલ VoIP એપ્લિકેશન
\item
  \textbf{બિઝનેસ PBX}: VoIP વાપરતી કોર્પોરેટ phone systems
\end{itemize}

\textbf{ફાયદાઓ:}

\begin{itemize}
\tightlist
\item
  \textbf{કિંમત અસરકારક}: લાંબા અંતરની કૉલ્સની ઓછી કિંમત
\item
  \textbf{ફીચર રિચ}: વિડિયો કૉલિંગ, કોન્ફરન્સિંગ, કૉલ ફોરવર્ડિંગ
\item
  \textbf{સ્કેલેબિલિટી}: નવા યુઝર્સ ઉમેરવા સરળ
\end{itemize}

\textbf{ગેરફાયદાઓ:}

\begin{itemize}
\tightlist
\item
  \textbf{ઇન્ટરનેટ ડિપેન્ડન્સી}: સ્થિર ઇન્ટરનેટ કનેક્શનની જરૂર
\item
  \textbf{ક્વોલિટી ઇશ્યુઝ}: network congestion થી સમસ્યા આવી શકે છે
\item
  \textbf{પાવર ડિપેન્ડન્સી}: પરંપરાગત ફોન્સ વિપરીત વીજળીની જરૂર
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``VoIP: ઇન્ટરનેટ પ્રોટોકોલ પર વૉઇસ''

\end{mnemonicbox}

communication માટે IP addresses માં translate કરે છે.

\item
  \textbf{નેમ સર્વર્સ}: DNS records સ્ટોર કરતા કમ્પ્યુટર્સ
\item
  \textbf{રિઝોલ્વર્સ}: DNS servers ને query કરતા ક્લાયન્ટ સોફ્ટવેર

\tightlist

\item
  સ્થાનિક resolver DNS server ને query કરે છે
\item
  DNS server અનુરૂપ IP address પરત કરે છે
\end{enumerate}

\end{solutionbox}
\begin{mnemonicbox}

\end{mnemonicbox}

\textbf{DSL વિષે ટૂંકી નોંધ લખો.}

\begin{solutionbox}
ડિજિટલ સબ્સ્ક્રાઇબર લાઇન વિદ્યમાન ટેલિફોન infrastructure પર
high-speed ઇન્ટરનેટ access પ્રદાન કરે છે.

\textbf{DSL ટેકનોલોજી:}

\begin{itemize}
\tightlist
\item
  \textbf{ફ્રીક્વન્સી ડિવિઝન}: voice કૉલ્સ કરતાં વધુ ફ્રીક્વન્સીઝ વાપરે છે
\item
  \textbf{સાથે ઉપયોગ}: ઇન્ટરનેટ અને ફોન એકસાથે કામ કરી શકે છે
\item
  \textbf{અંતર મર્યાદા}: exchange પાસેથી અંતર સાથે performance ઘટે છે
\end{itemize}

\textbf{DSL પ્રકારો:}

\begin{itemize}
\tightlist
\item
  \textbf{ADSL}: રહેવાસી યુઝર્સ માટે અસમમિત ઝડપ
\item
  \textbf{SDSL}: બિઝનેસ એપ્લિકેશન્સ માટે સમમિત ઝડપ
\item
  \textbf{VDSL}: ટૂંકા અંતર પર ખૂબ વધુ ઝડપ
\end{itemize}

\textbf{ફાયદાઓ:}

\begin{itemize}
\tightlist
\item
  \textbf{વિદ્યમાન ઇન્ફ્રાસ્ટ્રક્ચર}: વિદ્યમાન ટેલિફોન લાઇન્સ વાપરે છે
\item
  \textbf{હંમેશા ચાલુ}: સતત ઇન્ટરનેટ કનેક્શન
\item
  \textbf{કિંમત અસરકારક}: સમર્પિત લાઇન્સ કરતાં ઓછો ખર્ચ
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``DSL: ફોન લાઇન્સ પર ડિજિટલ સબ્સ્ક્રાઇબર લાઇન''

\end{mnemonicbox}

\end{itemize}

\end{solutionbox}

\item
  \textbf{HTTP પ્રોટોકોલ}: વેબ કમ્યુનિકેશન માટે હાઇપરટેક્સ્ટ ટ્રાન્સફર પ્રોટોકોલ

  \textbf{વેબ બ્રાઉઝર}: વેબ પેજીસ access કરવા માટે ક્લાયન્ટ સોફ્ટવેર
\item
  \textbf{વેબ સર્વર}: વેબસાઇટ્સ હોસ્ટ કરે છે અને વેબ પેજીસ સર્વ કરે છે
\end{itemize}

\textbf{FTP (ફાઇલ ટ્રાન્સફર પ્રોટોકોલ):}

\begin{itemize}
\tightlist
\item

\item
  \textbf{ક્લાયન્ટ-સર્વર}: FTP ક્લાયન્ટ FTP સર્વર સાથે જોડાય છે
\item
  \textbf{બે મોડ્સ}: ડેટા ટ્રાન્સફર માટે active અને passive મોડ્સ
\item
  \textbf{ઓથેન્ટિકેશન}: access control માટે યુઝરનેમ અને પાસવર્ડ
\end{itemize}

\textbf{લક્ષણો:}

\begin{itemize}
\tightlist
\item
  \textbf{WWW}: ગ્રાફિકલ ઇન્ટરફેસ, મલ્ટિમીડિયા સાપોર્ટ, હાઇપરલિંક્સ
\item
  \textbf{FTP}: મોટી ફાઇલ ટ્રાન્સફર, ડિરેક્ટરી નેવિગેશન, resume capability
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``WWW: વેબ વર્લ્ડ વાઇડ, FTP: ફાઇલ ટ્રાન્સફર પ્રોટોકોલ''

\item

\item
  \textbf{SSL/TLS}: સુરક્ષિત વેબ communications માટે બંને પ્રકારો વાપરે છે

\end{solutionbox}

\textbf{સાયબર સિક્યુરિટી ઉપર ટૂંક નોંધ લખો.}

\begin{solutionbox}
સાયબર સિક્યુરિટી ડિજિટલ attacks અને અનધિકૃત access પાસેથી
ડિજિટલ સિસ્ટમ્સ, નેટવર્ક્સ અને ડેટાને સુરક્ષિત કરે છે.

\textbf{મુખ્ય ઘટકો:}

\begin{itemize}
\tightlist
\item
  \textbf{નેટવર્ક સિક્યુરિટી}: intrusions પાસેથી નેટવર્ક infrastructure ને
  સુરક્ષિત કરે છે
\item
  \textbf{ડેટા પ્રોટેક્શન}: theft પાસેથી સંવેદનશીલ માહિતીને સુરક્ષિત રાખે છે
\item
  \textbf{એપ્લિકેશન સિક્યુરિટી}: vulnerabilities પાસેથી સોફ્ટવેર એપ્લિકેશન્સને
  સુરક્ષિત કરે છે
\end{itemize}

\textbf{સામાન્ય ધમકીઓ:}

\begin{itemize}
\tightlist
\item
  \textbf{મેલવેર}: સિસ્ટમ્સને નુકસાન પહોંચાડતા વાયરસ, worms, trojans
\item
  \textbf{ફિશિંગ}: credentials ચોરવાના કપટપૂર્ણ પ્રયાસો
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``સાયબર સિક્યુરિટી: ડિજિટલ અસ્કયામતોને સુરક્ષિત કરો''

\end{mnemonicbox}

\item
  \textbf{બ્લેક હેટ}: ગેરકાયદેસર હેતુઓ માટે દુર્ભાવનાપૂર્ણ હેકિંગ
\item
  \textbf{ગ્રે હેટ}: નૈતિક અને દુર્ભાવનાપૂર્ણ હેકિંગ વચ્ચે

\item
  \textbf{સોશિયલ એન્જિનિયરિંગ}: માહિતી પ્રગટ કરવા માટે લોકોને ચાલાકીથી
  પ્રભાવિત કરવું
\item
  \textbf{મેલવેર}: વાયરસ, trojans, ransomware
\item
  \textbf{નેટવર્ક એટેક્સ}: મેન-ઇન-ધ-મિડલ, પેકેટ સ્નિફિંગ
\end{itemize}

\textbf{સાવચેતીઓ:}

\item
  \textbf{નિયમિત અપડેટ્સ}: સોફ્ટવેર અને સિસ્ટમ્સને અપડેટ રાખો
\item
  \textbf{ફાયરવોલ}: અનધિકૃત access block કરવા માટે ફાયરવોલ વાપરો
\item
  \textbf{એન્ટીવાયરસ}: એન્ટીવાયરસ સોફ્ટવેર નિયમિત ઇન્સ્ટોલ અને અપડેટ કરો
\end{itemize}

\end{solutionbox}

\end{mnemonicbox}

\textbf{સંક્ષિપ્તમાં Information Technology (Amendment) Act 2008, અને ભારતમાં
સાયબર કાયદાઓ પર તેની અસરનું વર્ણન કરો.}

\begin{solutionbox}
IT સુધારા કાયદો 2008 એ ભારતના સાયબર કાયદા ફ્રેમવર્કને નોંધપાત્ર
રીતે મજબૂત બનાવ્યો અને સાયબર ક્રાઇમ કાયદાકીય વિસ્તારનો વિસ્તાર કર્યો.

\begin{itemize}
\tightlist
\item
  \textbf{ડેટા પ્રોટેક્શન}: સંવેદનશીલ વ્યક્તિગત ડેટાને સુરક્ષિત કરવા માટે વધારેલી
  જોગવાઈઓ
\item
  \textbf{સાયબર ક્રાઇમ વ્યાખ્યાઓ}: identity theft સહિત સાયબર ક્રાઇમની વિસ્તૃત
  વ્યાખ્યાઓ
\item
  \textbf{દંડ}: વિવિધ સાયબર અપરાધો માટે વધારેલા દંડ
\item
  \textbf{સાયબર આતંકવાદ}: સાયબર આતંકવાદ સાથે વ્યવહાર કરવા માટે જોગવાઈઓ દાખલ
  કરી
\end{itemize}

\textbf{મુખ્ય જોગવાઈઓ:}

\begin{itemize}
\tightlist
\item
  \textbf{કલમ 43A}: બેદરકારી માટે ડેટા પ્રોટેક્શન અને વળતર
\item
  \textbf{કલમ 66A}: આક્રામક સંદેશાઓ માટે સજા (બાદમાં રદ કરાઈ)

\item
  \textbf{કલમ 66D}: કમ્પ્યુટર રિસોર્સ વાપરીને વ્યક્તિત્વ દ્વારા છેતરપિંડી

\tightlist

\item
  \textbf{બિઝનેસ કમ્પ્લાયન્સ}: બિઝનેસ માટે ડેટા પ્રોટેક્શન પગલાં ફરજિયાત બનાવ્યા
\item
  \textbf{કાયદા અમલીકરણ}: તપાસ સાધનો સાથે સત્તાવાળાઓને સશક્ત બનાવ્યા
\item
  \textbf{આંતરરાષ્ટ્રીય સહયોગ}: સાયબર ક્રાઇમ તપાસમાં સહયોગને સરળ બનાવ્યું

\textbf{ડેટા પ્રોટેક્શન આવશ્યકતાઓ:}

\item
  \textbf{વાજબી સુરક્ષા}: કંપનીઓએ વાજબી સુરક્ષા પ્રથાઓ અમલમાં મૂકવી જોઈએ

  \textbf{ભંગ અધિસૂચના}: ડેટા ભંગની ફરજિયાત રિપોર્ટિંગ
\item

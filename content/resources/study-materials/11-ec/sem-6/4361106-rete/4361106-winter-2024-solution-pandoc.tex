\textbf{Explain with diagram Solar Photovoltaic effect \& Principle of
photovoltaic conversion.}

\begin{solutionbox}

\begin{verbatim}
      Sunlight (Photons)
            ↓
    ┌─────────────────┐
    │   N{-Type Layer  │ (Negative)}
    │  (Phosphorus)   │
    ├─────────────────┤  P{-N Junction}
    │   P{-Type Layer  │ (Positive)}
    │    (Boron)      │
    └─────────────────┘
            │
    ┌───────┴───────┐
    │   External    │
    │    Circuit    │
    └───────────────┘
\end{verbatim}

\textbf{Photovoltaic Effect Process}:

\begin{itemize}
\tightlist
\item
  \textbf{Photon Absorption}: Solar photons hit silicon atoms
\item
  \textbf{Electron Excitation}: Electrons gain energy and move to
  conduction band
\item
  \textbf{Charge Separation}: P-N junction creates electric field
\item
  \textbf{Current Flow}: Electrons flow through external circuit
\end{itemize}

\textbf{Key Parameters}:

\begin{itemize}
\tightlist
\item
  \textbf{Band Gap}: Energy difference between valence and conduction
  bands
\item
  \textbf{Open Circuit Voltage}: Maximum voltage when no current flows
\item
  \textbf{Short Circuit Current}: Maximum current when terminals are
  shorted
\end{itemize}

\textbf{Conversion Efficiency}:

\begin{itemize}
\tightlist
\item
  \textbf{Theoretical Maximum}: \textasciitilde33\% for single junction
  cells
\item
  \textbf{Practical Efficiency}: 15-22\% for commercial cells
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``Photons Push Electrons Past Junction - Power
Production Perfectly Planned''

\end{mnemonicbox}

\subsection*{પ્રશ્ન 1(બ) [4

\textbf{સર્ચ એન્જિન અલ્ગોરિધમનું કાર્ય સમજાવો.}

\begin{solutionbox}

સર્ચ એન્જિન અલ્ગોરિધમ સિસ્ટેમેટિક પ્રક્રિયાઓ દ્વારા સંબંધિત પરિણામો પહોંચાડે છે:

    A[વેબ ક્રોલિંગ] {-{-}{} B[ઇન્ડેક્સિંગ]}

    C {-{-}{} D[રેન્કિંગ અલ્ગોરિધમ]}

\begin{itemize}
\tightlist
\item
  \textbf{ક્રોલિંગ}: સર્ચ બોટ્સ નવી સામગ્રી શોધવા માટે વેબસાઇટ્સ સતત સ્કેન કરે છે
\item
  \textbf{ઇન્ડેક્સિંગ}: વિશ્લેષિત સામગ્રી કીવર્ડ્સ સાથે વિશાળ ડેટાબેઝમાં સંગ્રહિત થાય છે
\item
  \textbf{ક્વેરી મેચિંગ}: વપરાશકર્તાના સર્ચ શબ્દો ઇન્ડેક્સ્ડ સામગ્રી સાથે મેચ થાય છે
\item
  \textbf{રેન્કિંગ ફેક્ટર્સ}: સામગ્રીની સંબંધિતતા, અધિકાર અને વપરાશકર્તા અનુભવ
  પોઝિશન નક્કી કરે છે
\end{itemize}

\end{solutionbox}

``CIRR'' (ક્રોલ, ઇન્ડેક્સ, રેન્ક, રિઝલ્ટ્સ)

\end{mnemonicbox}

  \textbf{ખર્ચ અસરકારક}: પેઇડ Ads થી વિપરીત ઓર્ગેનિક રેન્કિંગ માટે સીધો ચુકવણી
  નથી
\item
  \textbf{વિશ્વાસ નિર્માણ}: ઉચ્ચ રેન્કિંગ વપરાશકર્તાઓ સાથે વિશ્વસનીયતા બનાવે છે
\item
  \textbf{લાંબા ગાળાના પરિણામો}: સમય સાથે ટકાઉ ટ્રાફિક વૃદ્ધિ
\end{itemize}

\subsection*{Question 3(a) [3 marks]}\label{q3a}

\textbf{Provide the definition of conversion rate and describe its
calculation.}

\begin{solutionbox}

\textbf{Conversion Rate Definition}: The percentage of website visitors
who complete a desired action (conversion) out of total visitors.

\textbf{Calculation Formula}:

\begin{verbatim}
Conversion Rate = (Number of Conversions / Total Visitors) \times 100
\end{verbatim}

\textbf{Example Calculation}:

\begin{itemize}
\tightlist
\item
  Total website visitors: 10,000
\item
  Number of purchases: 250
\item
  Conversion Rate = (250 \div 10,000) \times 100 = 2.5\%
\end{itemize}

\textbf{Types of Conversions}:

\begin{itemize}
\tightlist
\item
  \textbf{Macro Conversions}: Purchases, sign-ups, downloads
\item
  \textbf{Micro Conversions}: Email subscriptions, product views, cart
  additions
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``Conversions Count from Total Traffic'' (CCTT)

\end{mnemonicbox}

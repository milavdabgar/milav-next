\subsection*{Question 1(b) [4 marks]}\label{q1b}

\textbf{Explain working of search engine algorithm.}

\begin{solutionbox}

Search engine algorithms work through systematic processes to deliver
relevant results:

\textbf{Process Flow:}

\begin{center}
\textbf{Mermaid Diagram (Code)}
\begin{verbatim}
{Shaded}
{Highlighting}[]
graph LR
    A[Web Crawling] {-{-}{} B[Indexing]}
    B {-{-}{} C[Query Processing]}
    C {-{-}{} D[Ranking Algorithm]}
    D {-{-}{} E[SERP Display]}
{Highlighting}
{Shaded}
\end{verbatim}
\end{center}

\begin{itemize}
\tightlist
\item
  \textbf{Crawling}: Search bots scan websites continuously to discover
  new content
\item
  \textbf{Indexing}: Analyzed content is stored in massive databases
  with keywords
\item
  \textbf{Query matching}: User search terms are matched with indexed
  content
\item
  \textbf{Ranking factors}: Content relevance, authority, and user
  experience determine position
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``CIRR'' (Crawl, Index, Rank, Results)

\end{mnemonicbox}

\subsection*{Question 3(a) [3 marks]}\label{q3a}

\textbf{Explain any one Multi-touch attribution model with example.}

\begin{solutionbox}

\textbf{Linear Attribution Model} distributes credit equally across all
touchpoints in customer journey.

\textbf{Example Scenario:} Customer journey: Social Media Ad \rightarrow Email \rightarrow
Website Visit \rightarrow Purchase

\textbf{Credit Distribution:}

\begin{itemize}
\tightlist
\item
  Social Media Ad: 25\%
\item
  Email: 25\%\\
\item
  Website Visit: 25\%
\item
  Purchase Page: 25\%
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``EQUAL Credit for ALL Touches'' (Linear = Equal
distribution)

\end{mnemonicbox}

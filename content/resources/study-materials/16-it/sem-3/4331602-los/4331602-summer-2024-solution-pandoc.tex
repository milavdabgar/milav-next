\# Method 1: Using if{-elif{-}else}
if [ $num1 {-ge} $num2 ] \&\& [ $num1 {-ge} $num3 ]; then
    max=$num1
elif [ $num2 {-ge} $num1 ] \&\& [ $num2 {-ge} $num3 ]; then
    max=$num2

    max=$num3

echo "Maximum number is: $max"

if [ $num1 {-gt} $num2 ]; then
    if [ $num1 {-gt} $num3 ]; then
        echo "Maximum: $num1"
    else
        echo "Maximum: $num3"
    fi
else
    if [ $num2 {-gt} $num3 ]; then
        echo "Maximum: $num2"
    else
        echo "Maximum: $num3"
    fi
fi

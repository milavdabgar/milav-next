\subsubsection{\texorpdfstring{1. \textbf{`પરિસ્થિતિશાસ્ત્ર' અને `નિવસનતંત્ર'
ની વ્યાખ્યા
આપો.}}{1. `પરિસ્થિતિશાસ્ત્ર' અને `નિવસનતંત્ર' ની વ્યાખ્યા આપો.}}\label{uxaaauxab0uxab8uxaa5uxaa4uxab6uxab8uxaa4uxab0-uxa85uxaa8-uxaa8uxab5uxab8uxaa8uxaa4uxaa4uxab0-uxaa8-uxab5uxaafuxa96uxaaf-uxa86uxaaa.}

\begin{solutionbox}
\textbf{Ecology} એ જીવિત જીવોના તેમના પર્યાવરણ સાથેના સંબંધોનો
વૈજ્ઞાનિક અભ્યાસ છે. \textbf{Ecosystem} એ એકમ તરીકે કામ કરતા જીવો અને તેમના
ભૌતિક પર્યાવરણનો જૈવિક સમુદાય છે.

Ecosystem & જીવંત અને નિર્જીવ ઘટકોની પરસ્પર ક્રિયા & તળાવનું ecosystem \\

\begin{itemize}
\tightlist
\item
  \textbf{જૈવિક ઘટકો}: તંત્રમાં જીવંત જીવો
\item
  \textbf{અજૈવિક ઘટકો}: હવા, પાણી, માટી જેવા નિર્જીવ પરિબળો
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``દરેક ઘટક એકસાથે રહે છે'' (Ecology Creates Living

\subsubsection{\texorpdfstring{2. \textbf{`પ્રદૂષણ' અને `પ્રદૂષક' ની વ્યાખ્યા
આપો.}}{2. `પ્રદૂષણ' અને `પ્રદૂષક' ની વ્યાખ્યા આપો.}}\label{uxaaauxab0uxaa6uxab7uxaa3-uxa85uxaa8-uxaaauxab0uxaa6uxab7uxa95-uxaa8-uxab5uxaafuxa96uxaaf-uxa86uxaaa.}

\begin{solutionbox}
\textbf{Pollution} એ પર્યાવરણમાં હાનિકારક પદાર્થોનો પ્રવેશ છે જે
પ્રતિકૂળ અસરો લાવે છે. \textbf{Pollutant} એ કોઈપણ પદાર્થ છે જે વધારે માત્રામાં
હાજર હોય ત્યારે પ્રદૂષણ લાવે છે.

\endhead
\bottomrule\noalign{}
\endlastfoot
Pollution & પર્યાવરણીય દૂષણ & હવા, પાણી, માટી, અવાજ \\
Pollutant & હાનિકારક પદાર્થ & ભૌતિક, રાસાયણિક, જૈવિક \\
\end{longtable}
}

\begin{itemize}
\tightlist
\item
  \textbf{પ્રાથમિક પ્રદૂષકો}: સીધા વિસર્જિત પદાર્થો
\item
  \textbf{ગૌણ પ્રદૂષકો}: વાતાવરણમાં પ્રતિક્રિયાઓથી બનેલા
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``પ્રદૂષણ સમસ્યાઓ પેદા કરે છે'' (Pollution Produces
Problems)

\end{mnemonicbox}
\subsubsection{\texorpdfstring{3. \textbf{`અવાજનું પ્રદૂષણ' એટલે શું? ધ્વનિની
તીવ્રતાનો એકમ શું
છે?}}{3. `અવાજનું પ્રદૂષણ' એટલે શું? ધ્વનિની તીવ્રતાનો એકમ શું છે?}}\label{uxa85uxab5uxa9cuxaa8-uxaaauxab0uxaa6uxab7uxaa3-uxa8fuxa9fuxab2-uxab6-uxaa7uxab5uxaa8uxaa8-uxaa4uxab5uxab0uxaa4uxaa8-uxa8fuxa95uxaae-uxab6-uxa9b}

\begin{solutionbox}
\textbf{Noise pollution} એ અનિચ્છિત અથવા વધુ પડતો અવાજ છે જે
માનવીય પ્રવૃત્તિઓને ખલેલ પહોંચાડે છે. ધ્વનિની તીવ્રતાનો એકમ \textbf{decibel (dB)}
છે.

30-40 dB & પુસ્તકાલય & આરામદાયક \\

90+ dB & ઉદ્યોગ & હાનિકારક \\

}

\tightlist

\item
  \textbf{પીડાની સીમા}: 120 dB
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``Decibel નુકસાન નક્કી કરે છે'' (dB Determines Damage)

\end{mnemonicbox}
\subsubsection{\texorpdfstring{4. \textbf{ઘન કચરાનું વ્યવસ્થાપન શું છે? તેના હેતુઓ
જણાવો.}}{4. ઘન કચરાનું વ્યવસ્થાપન શું છે? તેના હેતુઓ જણાવો.}}\label{uxa98uxaa8-uxa95uxa9auxab0uxaa8-uxab5uxaafuxab5uxab8uxaa5uxaaauxaa8-uxab6-uxa9b-uxaa4uxaa8-uxab9uxaa4uxa93-uxa9cuxaa3uxab5.}

\begin{solutionbox}
\textbf{Solid waste management} એ પર્યાવરણીય અસર ઘટાડવા અને
જાહેર આરોગ્યની સુરક્ષા માટે કચરાના ઉત્પાદનથી અંતિમ નિકાલ સુધીનું વ્યવસ્થિત સંચાલન છે.

\textbf{હેતુઓ:}

\begin{itemize}
\tightlist
\item
  \textbf{જાહેર આરોગ્ય સંરક્ષણ}: રોગ પ્રસારણ અટકાવવું
\item
  \textbf{પર્યાવરણ સંરક્ષણ}: પ્રદૂષણ અને દૂષણ ઘટાડવું
\item
  \textbf{સંસાધન પુનઃપ્રાપ્તિ}: સામગ્રીનું પુનઃઉપયોગ અને રીસાયકલિંગ
\item
  \textbf{ખર્ચ અસરકારકતા}: આર્થિક કચરા નિયંત્રણ
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``લોકો સંસાધન સંરક્ષણની અપેક્ષા રાખે છે'' (Protection,
Environment, Resource, Cost)

\end{mnemonicbox}

\subsubsection{\texorpdfstring{6. \textbf{`આબોહવા (જલવાયુ) પરિવર્તન' શું
છે?}}{6. `આબોહવા (જલવાયુ) પરિવર્તન' શું છે?}}\label{uxa86uxaacuxab9uxab5-uxa9cuxab2uxab5uxaaf-uxaaauxab0uxab5uxab0uxaa4uxaa8-uxab6-uxa9b}

\begin{solutionbox}
\textbf{Climate change} એ મુખ્યત: માનવીય પ્રવૃત્તિઓ અને
greenhouse gas ઉત્સર્જનને કારણે વૈશ્વિક તાપમાન અને હવામાન પેટર્નમાં લાંબા ગાળાના
ફેરફારોનો સંદર્ભ આપે છે.

\textbf{કારણો:}

\begin{itemize}
\tightlist
\item
  \textbf{Greenhouse gases}: CO₂, CH₄, N₂O ઉત્સર્જન
\item
  \textbf{વનનાશ}: કાર્બન શોષણમાં ઘટાડો
\item
  \textbf{ઔદ્યોગિક પ્રવૃત્તિઓ}: અશ્મિભૂત ઇંધનનું બર્નિંગ
\end{itemize}

\textbf{અસરો:}

\begin{itemize}
\tightlist
\item
  \textbf{વધતું તાપમાન}: વૈશ્વિક ઉષ્ણતા
\item
  \textbf{દરિયાઈ સ્તરમાં વધારો}: બરફ પીગળવાથી
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``પરિવર્તન પરિણામો બનાવે છે'' (Change Creates
Consequences)

\end{mnemonicbox}
\subsubsection{\texorpdfstring{7. \textbf{C.F.C શું
છે?}}{7. C.F.C શું છે?}}\label{c.f.c-uxab6-uxa9b}

\begin{solutionbox}
\textbf{CFC (Chlorofluorocarbon)} એ કાર્બન, ફ્લોરિન અને
ક્લોરિન અણુઓ ધરાવતા કૃત્રિમ સંયોજનો છે, જે અગાઉ refrigeration અને aerosols માં
વપરાતા હતા.

\textbf{ગુણધર્મો:}

\begin{itemize}
\tightlist
\item
  \textbf{ઓઝોન નાશક}: stratospheric ozone નાશ કરે છે
\item
  \textbf{Greenhouse gas}: વૈશ્વિક ઉષ્ણતામાં યોગદાન
\item
  \textbf{સ્થિર સંયોજનો}: લાંબા વાતાવરણીય આયુષ્ય
\item
  \textbf{Montreal Protocol}: આંતરરાષ્ટ્રીય પ્રતિબંધ કરાર
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``ક્લોરિન ફ્લોરિન કાર્બન'' (CFC ઘટકો)

\end{mnemonicbox}
\subsubsection{\texorpdfstring{8. \textbf{ISO-14000 ના ફાયદા
આપો.}}{8. ISO-14000 ના ફાયદા આપો.}}\label{iso-14000-uxaa8-uxaabuxaafuxaa6-uxa86uxaaa.}

\begin{solutionbox}
\textbf{ISO 14000} પર્યાવરણીય વ્યવસ્થાપન પ્રણાલીઓ માટેનું
આંતરરાષ્ટ્રીય ધોરણ છે.

\textbf{ફાયદા:}

\begin{itemize}
\tightlist
\item
  \textbf{પર્યાવરણીય અનુપાલન}: કાનૂની જરૂરિયાતોની પૂર્તિ
\item
  \textbf{ખર્ચ ઘટાડો}: કુશળ સંસાધન ઉપયોગ
\item
  \textbf{બજાર ફાયદો}: કંપનીની છબીમાં સુધારો
\item
  \textbf{જોખમ વ્યવસ્થાપન}: પર્યાવરણીય દુર્ઘટનાઓ અટકાવવી
\end{itemize}

\subsubsection{\texorpdfstring{9. \textbf{ભારતમાં પર્યાવરણ સંબંધિત વિવિધ
કાયદાઓની યાદી
બનાવો.}}{9. ભારતમાં પર્યાવરણ સંબંધિત વિવિધ કાયદાઓની યાદી બનાવો.}}\label{uxaaduxab0uxaa4uxaae-uxaaauxab0uxaafuxab5uxab0uxaa3-uxab8uxaacuxaa7uxaa4-uxab5uxab5uxaa7-uxa95uxaafuxaa6uxa93uxaa8-uxaafuxaa6-uxaacuxaa8uxab5.}

\begin{solutionbox}
ભારતમાં વ્યાપક પર્યાવરણીય કાયદાકીય માળખું છે.

\textbf{મુખ્ય કાયદાઓ:}

\begin{itemize}
\tightlist
\item
  \textbf{Air Act (1981)}: હવા પ્રદૂષણ નિયંત્રણ
\item
  \textbf{Water Act (1974)}: પાણી પ્રદૂષણ અટકાવવા
\item
  \textbf{Environment Protection Act (1986)}: વ્યાપક પર્યાવરણીય કાયદો
\item
  \textbf{Wildlife Protection Act (1972)}: જૈવવિવિધતા સંરક્ષણ
\item
  \textbf{Forest Conservation Act (1980)}: વન સંરક્ષણ
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``તમામ પાણી પર્યાવરણ વન્યજીવ વન'' (AWEWF)

\end{mnemonicbox}
\subsubsection{\texorpdfstring{10. \textbf{વરસાદના પાણીના સંચયની વિવિધ
પદ્ધતિઓની યાદી
બનાવો.}}{10. વરસાદના પાણીના સંચયની વિવિધ પદ્ધતિઓની યાદી બનાવો.}}\label{uxab5uxab0uxab8uxaa6uxaa8-uxaaauxaa3uxaa8-uxab8uxa9auxaafuxaa8-uxab5uxab5uxaa7-uxaaauxaa6uxaa7uxaa4uxa93uxaa8-uxaafuxaa6-uxaacuxaa8uxab5.}

\begin{solutionbox}

\begin{solutionbox}

             |

\textbf{ઘટકો:}

\begin{itemize}
\tightlist
\item
  \textbf{ડાયજેસ્ટર ટેન્ક}: Anaerobic fermentation ચેમ્બર
\item
  \textbf{ગેસ ડોમ}: બાયોગેસ સંગ્રહ અને સ્ટોરેજ
\item
  \textbf{ઇનલેટ પાઇપ}: કચરા સામગ્રીનું ફીડિંગ
\item
  \textbf{આઉટલેટ પાઇપ}: પચેલા સ્લરીને દૂર કરવું
\end{itemize}

\textbf{પ્રક્રિયા:}

\begin{itemize}
\tightlist
\item
  \textbf{હાઇડ્રોલિસિસ}: જટિલ કાર્બનિક પદાર્થો તૂટે છે
\item
  \textbf{એસિડોજેનેસિસ}: એસિડ બનાવતા બેક્ટેરિયાની ક્રિયા
\item

\item

\end{itemize}

\begin{itemize}
\tightlist
\item
  \textbf{તાપમાન}: 30-40°C શ્રેષ્ઠ
\item
  \textbf{pH}: 6.8-7.2 રેન્જ
\item
  \textbf{રીટેન્શન ટાઇમ}: 15-30 દિવસ
\item
  \textbf{C:N રેશિયો}: 20-30:1 શ્રેષ્ઠ
\end{itemize}

\begin{itemize}
\tightlist
\item
  \textbf{રસોઈ ઇંધન}: ઘરેલું ઊર્જાની જરૂરિયાતો
\item
  \textbf{પ્રકાશ}: ગેસ લેમ્પ રોશની
\item
  \textbf{વીજળી}: જનરેટર પાવર
\item
  \textbf{ખાતર}: પોષક તત્વોથી સમૃદ્ધ સ્લરી
\end{itemize}

\begin{itemize}
\tightlist
\item
  \textbf{નવીકરણીય ઊર્જા}: ટકાઉ ઇંધન સ્ત્રોત
\item
  \textbf{કચરા વ્યવસ્થાપન}: કાર્બનિક કચરાનો ઉપયોગ
\item
  \textbf{પર્યાવરણીય ફાયદા}: મિથેન ઉત્સર્જનમાં ઘટાડો
\item
  \textbf{આર્થિક ફાયદા}: ઇંધન પર ખર્ચ બચત
\end{itemize}

\end{solutionbox}

\subsubsection{\texorpdfstring{\textbf{`ગ્લોબલ વોર્મિંગ' પર ટૂંક નોંધ

\begin{solutionbox}
\textbf{Global warming} માનવીય પ્રવૃત્તિઓને કારણે પૃથ્વીના સરેરાશ

\begin{itemize}
\tightlist
\item
  \textbf{Greenhouse gases}: CO₂, CH₄, N₂O ઉત્સર્જન
\item
  \textbf{વનનાશ}: કાર્બન શોષણમાં ઘટાડો
\item
  \textbf{ઔદ્યોગિક પ્રવૃત્તિઓ}: અશ્મિભૂત ઇંધન દહન
\item
  \textbf{પરિવહન}: વાહન ઉત્સર્જન
\end{itemize}

\textbf{અસરો:}

\begin{itemize}
\tightlist
\item
  \textbf{તાપમાન વૃદ્ધિ}: પૂર્વ-ઔદ્યોગિક સમયથી 1.1°C
\item
  \textbf{બરફ પીગળવો}: આર્કટિક દરિયાઈ બરફ, ગ્લેશિયર સંકુચિત થવા
\item
  \textbf{દરિયાઈ સ્તર વૃદ્ધિ}: દરિયાકાંઠાના પૂરનું જોખમ
\item
  \textbf{હવામાન ફેરફાર}: ચરમ ઘટનાઓની આવૃત્તિ
\end{itemize}

\textbf{પુરાવા:}

\begin{itemize}
\tightlist
\item
  \textbf{તાપમાનના રેકોર્ડ}: તાજેતરના દાયકાઓમાં સૌથી ગરમ વર્ષો
\item
  \textbf{બરફ કોર ડેટા}: ઐતિહાસિક CO₂ સ્તરો
\item
  \textbf{સેટેલાઇટ માપ}: વૈશ્વિક તાપમાન મોનિટરિંગ
\end{itemize}

\begin{itemize}
\tightlist
\item
  \textbf{નવીકરણીય ઊર્જા}: સ્વચ્છ પાવર સ્ત્રોતો
\item
  \textbf{ઊર્જા કાર્યક્ષમતા}: ઘટતો વપરાશ

\item
  \textbf{આંતરરાષ્ટ્રીય સહયોગ}: પેરિસ એગ્રીમેન્ટ
\end{itemize}

\end{solutionbox}

\begin{solutionbox}
\textbf{5R concept} ટકાઉ સંસાધન ઉપયોગ માટે કચરા વ્યવસ્થાપન

\textbf{5 R's:}

\textbf{1. Refuse:}

\begin{itemize}
\tightlist
\item
  \textbf{બિનજરૂરી વસ્તુઓ ટાળો}: એકવાર વપરાશની વસ્તુઓને ના કહો
\item
  \textbf{ઉદાહરણો}: પ્લાસ્ટિક બેગ, સ્ટ્રો, વધુ પડતું પેકેજિંગ
\end{itemize}

\textbf{2. Reduce:}

\begin{itemize}
\tightlist
\item
  \textbf{વપરાશ ઓછો કરો}: ઓછા સંસાધનોનો ઉપયોગ
\item
  \textbf{ઉદાહરણો}: ઊર્જા સંરક્ષણ, પાણી બચાવવું
\end{itemize}

\textbf{3. Reuse:}

\begin{itemize}
\tightlist
\item
  \textbf{અનેક વાર ઉપયોગ}: ઉત્પાદનનું જીવન વધારવું
\item
  \textbf{ઉદાહરણો}: કાચના જાર કન્ટેનર તરીકે, કાગળ બંને બાજુ
\end{itemize}

\textbf{4. Repurpose:}

\begin{itemize}
\tightlist
\item
  \textbf{સર્જનાત્મક પુનઃઉપયોગ}: જૂની વસ્તુઓ માટે નવું કાર્ય
\item
  \textbf{ઉદાહરણો}: ટાયર પ્લાન્ટર, બોટલ પક્ષી ફીડર
\end{itemize}

\textbf{5. Recycle:}

\begin{itemize}
\tightlist
\item
  \textbf{સામગ્રી પુનઃપ્રાપ્તિ}: નવા ઉત્પાદનોમાં પ્રક્રિયા
\item
  \textbf{ઉદાહરણો}: કાગળ, પ્લાસ્ટિક, ધાતુ રીસાયક્લિંગ
\end{itemize}

\textbf{ફાયદા:}

\begin{itemize}
\tightlist
\item
  \textbf{કચરા ઘટાડો}: લેન્ડફિલ પર ઓછો બોજ
\item
  \textbf{સંસાધન સંરક્ષણ}: કુદરતી સંસાધન સાચવણી
\item
  \textbf{ખર્ચ બચત}: આર્થિક ફાયદા
\item
  \textbf{પર્યાવરણ સંરક્ષણ}: પ્રદૂષણ ઘટાડો
\end{itemize}

\end{solutionbox}

\end{mnemonicbox}

\begin{solutionbox}
\textbf{ભારતમાં પર્યાવરણીય કાયદાઓ:}

\begin{itemize}
\tightlist
\item
  \textbf{Water (Prevention and Control of Pollution) Act, 1974}
\item
  \textbf{Air (Prevention and Control of Pollution) Act, 1981}
\item
  \textbf{Environment Protection Act, 1986}
\item
  \textbf{Wildlife Protection Act, 1972}
\item
  \textbf{Forest (Conservation) Act, 1980}
\item
  \textbf{Biodiversity Act, 2002}
\end{itemize}

\textbf{Environment Protection Act, 1986:} \textbf{હેતુઓ:}

\begin{itemize}
\tightlist
\item
  \textbf{વ્યાપક માળખું}: એકંદર પર્યાવરણ સંરક્ષણ
\item
  \textbf{પ્રદૂષણ નિવારણ}: હવા, પાણી, માટી દૂષણ નિયંત્રણ
\item
  \textbf{માનદંડ સેટિંગ}: પર્યાવરણીય ગુણવત્તા માનદંડો
\item
  \textbf{અમલીકરણ}: ઉલ્લંઘન માટે દંડ
\end{itemize}

\textbf{શક્તિઓ:}

\begin{itemize}
\tightlist
\item
  \textbf{કેન્દ્ર સરકાર સત્તા}: પર્યાવરણીય નિયમો
\item
  \textbf{નિરીક્ષણ અધિકારો}: ઔદ્યોગિક સુવિધાઓની દેખરેખ
\item
  \textbf{બંધ કરવાના આદેશો}: બિન-અનુપાલન કરતા ઉદ્યોગો
\item
  \textbf{કટોકટીના પગલાં}: પર્યાવરણીય સંકટોનો પ્રતિસાદ
\end{itemize}

\textbf{મહત્વ:}

\begin{itemize}
\tightlist
\item
  \textbf{છત્ર કાયદો}: બધા પર્યાવરણીય પાસાઓને આવરે છે
\item
  \textbf{ભોપાલ દુર્ઘટના પછી}: ઔદ્યોગિક અકસ્માતોનો પ્રતિસાદ
\end{itemize}

\end{solutionbox}

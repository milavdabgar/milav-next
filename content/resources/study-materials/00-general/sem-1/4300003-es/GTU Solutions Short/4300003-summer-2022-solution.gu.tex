\questionmarks{1}{a}{3}
\textbf{ટૂંકનોધ લખો: પારિસ્થિતિક પિરામિડ.}

\begin{solutionbox}
    \begin{answertable}{પારિસ્થિતિક પિરામિડના પ્રકારો}
    \begin{tabulary}{\linewidth}{L L L}
        \toprule
        \textbf{પ્રકાર} & \textbf{વર્ણન} & \textbf{ઉદાહરણ} \\
        \midrule
        \textbf{સંખ્યાનો પિરામિડ} & દરેક સ્તરે જીવોની સંખ્યા દર્શાવે છે & વૃક્ષો $\rightarrow$ જંતુઓ $\rightarrow$ પક્ષીઓ \\
        \textbf{બાયોમાસ પિરામિડ} & જીવોનું કુલ દળ દર્શાવે છે & ઉત્પાદક સ્તરે વધુ \\
        \textbf{ઊર્જા પિરામિડ} & સ્તરોમાં ઊર્જાનો પ્રવાહ દર્શાવે છે & હંમેશા સીધો \\
        \bottomrule
    \end{tabulary}
    \end{answertable}

    \begin{itemize}
        \item \textbf{ઊર્જા સ્થાનાંતરણ}: માત્ર 10\% ઊર્જા આગલા સ્તરમાં જાય છે
        \item \textbf{પોષક સ્તરો}: ઉત્પાદકો, પ્રાથમિક ઉપભોક્તાઓ, ગૌણ ઉપભોક્તાઓ
        \item \textbf{હંમેશા સીધો}: ઊર્જા પિરામિડ ક્યારેય ઊંધો નથી થતો
    \end{itemize}

    \begin{mnemonicbox}સંખ્યા-બાયોમાસ-ઊર્જા ઉપર વહે છે\end{mnemonicbox}
\end{solutionbox}

\questionmarks{1}{b}{4}
\textbf{વૈશ્વિક પારિસ્થિતિકીય ઊછાળ વિશે ટૂંકનોધ લખો.}

    \begin{answertable}{વૈશ્વિક પારિસ્થિતિકીય ઊછાળના ઘટકો}
    \begin{tabulary}{\linewidth}{L L}

        \textbf{પરિબળ} & \textbf{વર્ણન} \\
        \midrule
        \textbf{પૃથ્વી ઓવરશૂટ દિવસ} & જે દિવસે વાર્ષિક સંસાધન વપરાશ પુનઃઉત્પાદન કરતાં વધે છે \\
        \textbf{પારિસ્થિતિક પદચિહ્ન} & કુદરતી સંસાધનો પર માનવીય માંગ \\
        \textbf{બાયોકેપેસિટી} & સંસાધનો પુનઃઉત્પન્ન કરવાની પૃથ્વીની ક્ષમતા \\
        \bottomrule
    \end{tabulary}
    \end{answertable}

    \begin{itemize}
        \item \textbf{હાલની સ્થિતિ}: વાર્ષિક 1.7 પૃથ્વી જેટલા સંસાધનોનો ઉપયોગ
        \item \textbf{પરિણામો}: હવામાન પરિવર્તન, જૈવવિવિધતા નુકસાન, સંસાધન અવક્ષય
        \item \textbf{ઉકેલો}: ટકાઉ વપરાશ, નવીકરણીય ઊર્જા અપનાવવી
    \end{itemize}

    \begin{mnemonicbox}માંગ પુરવઠા કરતાં વધારે = ઊછાળ\end{mnemonicbox}
\end{solutionbox}

\questionmarks{1}{c}{7}

        \draw [gtu arrow] (decomp) -| (atmos);
        \draw [gtu arrow] (atmos) -- (ocean);
        \draw [gtu arrow] (ocean) -- (atmos);
    \end{tikzpicture}
    \end{center}

    \begin{answertable}{નાઇટ્રોજન ચક્રના તબક્કાઓ}
    \begin{tabulary}{\linewidth}{L L L}
        \toprule
        \textbf{તબક્કો} & \textbf{પ્રક્રિયા} & \textbf{જીવતંત્ર} \\
        \midrule
        \textbf{નાઇટ્રોજન સ્થિરીકરણ} & N$_2$ $\rightarrow$ NH$_3$ & રાયઝોબિયમ બેક્ટેરિયા \\
        \textbf{નાઇટ્રિફિકેશન} & NH$_3$ $\rightarrow$ NO$_3$ & નાઇટ્રોસોમોનાસ, નાઇટ્રોબેક્ટર \\
        \textbf{ડિનાઇટ્રિફિકેશન} & NO$_3$ $\rightarrow$ N$_2$ & ડિનાઇટ્રિફાઇંગ બેક્ટેરિયા \\
        \bottomrule
    \end{tabulary}
    \end{answertable}

    \begin{itemize}
        \item \textbf{મહત્વ}: પ્રોટીન સંશ્લેષણ અને DNA રચના માટે આવશ્યક
        \item \textbf{માનવીય અસર}: ખાતરો કુદરતી સંતુલન વિખેરે છે
        \item \textbf{સંરક્ષણ}: રાસાયણિક ખાતરનો ઉપયોગ ઘટાડવો
    \end{itemize}

    \begin{mnemonicbox}બેક્ટેરિયા નાઇટ્રોજન ઠીક કરે છે, છોડ વાપરે છે\end{mnemonicbox}
\end{solutionbox}

    \end{itemize}

\end{solutionbox}

\questionmarks{5}{c}{7}
\textbf{5R નો કોન્સેપ્ટ સમજાવો.}

\begin{solutionbox}
    \textbf{5R કોન્સેપ્ટ:}
    પર્યાવરણીય ટકાઉપણા માટે કચરા વ્યવસ્થાપન શ્રેણી.

    \textbf{5Rs:}
    \begin{center}
    \begin{tikzpicture}[node distance=1.5cm, auto]
        \node (r5) [gtu block] {5R શ્રેણી};
        \node (refuse) [gtu block, below=of r5] {1. ઇનકાર};
        \node (reduce) [gtu block, below=of refuse] {2. ઘટાડો};
        \node (reuse) [gtu block, below=of reduce] {3. પુનઃઉપયોગ};
        \node (repurpose) [gtu block, below=of reuse] {4. પુનઃહેતુ};
        \node (recycle) [gtu block, below=of repurpose] {5. પુનર્ચક્રણ};

        \draw [gtu arrow] (r5) -- (refuse);
        \draw [gtu arrow] (refuse) -- (reduce);
        \draw [gtu arrow] (reduce) -- (reuse);
        \draw [gtu arrow] (reuse) -- (repurpose);
        \draw [gtu arrow] (repurpose) -- (recycle);
    \end{tikzpicture}
    \end{center}

    \begin{answertable}{5R વિગતવાર સમજાવટ}
    \begin{tabulary}{\linewidth}{L L L L}
        \toprule
        \textbf{R} & \textbf{વ્યાખ્યા} & \textbf{ઉદાહરણો} & \textbf{ફાયદા} \\
        \midrule
        \textbf{ઇનકાર (Refuse)} & બિનજરૂરી વસ્તુઓ ટાળવી & પ્લાસ્ટિક બેગ, ડિસ્પોઝેબલ & કચરા ઉત્પાદન અટકાવવું \\
        \textbf{ઘટાડવું (Reduce)} & વપરાશ ન્યૂનીકરણ & ઊર્જા, પાણી, સામગ્રી & સંસાધનની માંગ ઓછી કરવી \\
        \textbf{પુનઃઉપયોગ (Reuse)} & વસ્તુઓનો વારંવાર ઉપયોગ & કન્ટેનર, કપડાં & ઉત્પાદનનું જીવન લંબાવવું \\
        \textbf{પુનઃહેતુ (Repurpose)} & વસ્તુઓ માટે નવા ઉપયોગ શોધવા & ટાયર પ્લાન્ટર, બોટલ હસ્તકલા & સર્જનાત્મક કચરો દિવર્ટ કરવું \\
        \textbf{પુનર્ચક્રણ (Recycle)} & નવા ઉત્પાદનોમાં પ્રક્રિયા કરવી & કાગળ, પ્લાસ્ટિક, ધાતુઓ & સામગ્રી પુનઃપ્રાપ્તિ \\
        \bottomrule
    \end{tabulary}
    \end{answertable}

    \textbf{અમલીકરણ વ્યૂહરચનાઓ:}
    
    \textbf{વ્યક્તિગત સ્તરે:}
    \begin{itemize}
        \item \textbf{ઇનકાર}: સિંગલ-યુઝ પ્લાસ્ટિકને ના કહો
        \item \textbf{ઘટાડો}: માત્ર જરૂરી વસ્તુઓ ખરીદો
        \item \textbf{પુનઃઉપયોગ}: કન્ટેનર અને સામગ્રીનો પુનઃઉપયોગ કરો
        \item \textbf{પુનઃહેતુ}: સર્જનાત્મક DIY પ્રોજેક્ટ્સ
        \item \textbf{પુનર્ચક્રણ}: યોગ્ય વર્ગીકરણ અને નિકાલ
    \end{itemize}

    \textbf{સમુદાય સ્તરે:}
    \begin{itemize}
        \item \textbf{જાગૃતિ કાર્યક્રમો}: 5R સિદ્ધાંતો વિશે શિક્ષણ
        \item \textbf{ઇન્ફ્રાસ્ટ્રક્ચર}: રીસાયક્લિંગ સુવિધાઓ અને સંગ્રહ સિસ્ટમ
        \item \textbf{નીતિઓ}: કચરા ઘટાડવાને પ્રોત્સાહન આપનાર નિયમો
        \item \textbf{પ્રોત્સાહન}: ટકાઉ પ્રથાઓ માટે પુરસ્કારો
    \end{itemize}

    \textbf{ઔદ્યોગિક સ્તરે:}
    \begin{itemize}
        \item \textbf{ટકાઉપણા માટે ડિઝાઇન}: લાંબા સમય સુધી ચાલતા ઉત્પાદનો
        \item \textbf{સામગ્રી પસંદગી}: રીસાયકલ અને બાયોડિગ્રેડેબલ સામગ્રી
        \item \textbf{પરિપત્ર અર્થવ્યવસ્થા}: બંધ-લૂપ ઉત્પાદન સિસ્ટમ
        \item \textbf{વિસ્તૃત ઉત્પાદક જવાબદારી}: ઉત્પાદક જવાબદારી
    \end{itemize}

    \textbf{પર્યાવરણીય ફાયદા:}
    \begin{itemize}
        \item \textbf{સંસાધન સંરક્ષણ}: ઘટેલી કાચી સામગ્રી નિષ્કર્ષણ
        \item \textbf{ઊર્જા બચત}: ઓછી ઉત્પાદન ઊર્જા જરૂરિયાતો
        \item \textbf{પ્રદૂષણ ઘટાડો}: ઘટેલું કચરો ઉત્પાદન
        \item \textbf{હવામાન સંરક્ષણ}: ઘટેલું ગ્રીનહાઉસ ગેસ ઉત્સર્જન
    \end{itemize}

    \textbf{આર્થિક ફાયદા:}
    \begin{itemize}
        \item \textbf{ખર્ચ બચત}: ઓછો નિકાલ અને સામગ્રી ખર્ચ
        \item \textbf{નોકરી સર્જન}: રીસાયક્લિંગ અને પુનઃઉપયોગ ક્ષેત્રોમાં ગ્રીન જોબ્સ
        \item \textbf{નવીનતા}: ટકાઉ તકનીકોનો વિકાસ
        \item \textbf{બજાર તકો}: નવા બિઝનેસ મોડેલ
    \end{itemize}

    \textbf{સામાજિક ફાયદા:}
    \begin{itemize}
        \item \textbf{સમુદાય સંલગ્નતા}: સામૂહિક પર્યાવરણીય ક્રિયા
        \item \textbf{આરોગ્ય સુધારણા}: સ્વચ્છ પર્યાવરણ
        \item \textbf{શિક્ષણ}: પર્યાવરણીય જાગૃતિ અને જવાબદારી
        \item \textbf{સાંસ્કૃતિક પરિવર્તન}: ટકાઉ જીવનશૈલી અપનાવવી
    \end{itemize}

    \textbf{પડકારો:}
    \begin{itemize}
        \item \textbf{વર્તન પરિવર્તન}: વપરાશની આદતો પર કાબુ મેળવવો
        \item \textbf{ઇન્ફ્રાસ્ટ્રક્ચર}: પર્યાપ્ત રીસાયક્લિંગ સુવિધાઓ
        \item \textbf{આર્થિક અવરોધો}: પ્રારંભિક રોકાણની જરૂરિયાતો
        \item \textbf{નીતિ સમર્થન}: સરકારી નિયમો અને પ્રોત્સાહન
    \end{itemize}

    \textbf{સફળતાની વાર્તાઓ:}
    \begin{itemize}
        \item \textbf{ઝીરો વેસ્ટ શહેરો}: સાન ફ્રાન્સિસ્કો, કામિકાત્સુ
        \item \textbf{કોર્પોરેટ પહેલ}: કંપની 5R કાર્યક્રમો
        \item \textbf{શાળા કાર્યક્રમો}: વિદ્યાર્થી પર્યાવરણીય શિક્ષણ
        \item \textbf{સમુદાય પ્રોજેક્ટ્સ}: સ્થાનિક કચરા ઘટાડવાના પ્રયાસો
    \end{itemize}

    \begin{mnemonicbox}ખરેખર ઘટાડો પુનઃઉપયોગ પુનઃહેતુ પુનર્ચક્રણ\end{mnemonicbox}
\end{solutionbox}

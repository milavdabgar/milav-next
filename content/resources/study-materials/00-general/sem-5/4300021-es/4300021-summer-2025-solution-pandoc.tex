\subsection*{Question 3(c) [7 marks]}\label{q3c}

\textbf{Explain Break-even analysis for a business practice.}

\begin{solutionbox}

\textbf{Break-even Formula:} Break-even Point = Fixed Costs \div (Selling
Price per unit - Variable Cost per unit)

\begin{center}
\textbf{Mermaid Diagram (Code)}
\begin{verbatim}
{Shaded}
{Highlighting}[]
graph LR
    A[Fixed Costs] {-{-}{} B[Variable Costs]}
    B {-{-}{} C[Total Costs]}
    C {-{-}{} D[Revenue Line]}
    D {-{-}{} E[Break{-}even Point]}
{Highlighting}
{Shaded}
\end{verbatim}
\end{center}

\textbf{Components:}

\begin{itemize}
\tightlist
\item
  \textbf{Fixed Costs}: Rent, salaries, insurance
\item
  \textbf{Variable Costs}: Raw materials, labor
\item
  \textbf{Selling Price}: Revenue per unit
\item
  \textbf{Contribution Margin}: Price minus variable cost
\end{itemize}

\textbf{Benefits:}

\begin{itemize}
\tightlist
\item
  \textbf{Profit Planning}: Determines minimum sales needed
\item
  \textbf{Price Setting}: Helps in pricing decisions
\item
  \textbf{Cost Control}: Identifies cost structure
\item
  \textbf{Investment Decisions}: Evaluates project viability
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``Fixed Variable Selling Contribution makes Profit
Price Cost Investment''

\end{mnemonicbox}
\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\textbf{What is project planning? Write down its purposes.}

\begin{solutionbox}

\textbf{Project Planning Definition:} The process of defining project
scope, objectives, and steps needed to achieve goals.

\textbf{Purposes:}

\begin{itemize}
\tightlist
\item
  \textbf{Goal Setting}: Clear objective definition
\item
  \textbf{Resource Allocation}: Efficient use of resources
\item
  \textbf{Time Management}: Schedule development
\item
  \textbf{Risk Mitigation}: Problem anticipation
\item
  \textbf{Quality Assurance}: Standard maintenance
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``Goals Resources Time Risk Quality''

\end{mnemonicbox}

\subsection*{Question 5(c) [7 marks]}\label{q5c}

\textbf{What is feasibility analysis in business? Explain various
analysis covered in feasibility study.}

\begin{solutionbox}

\textbf{Feasibility Analysis Definition:} Systematic evaluation of
project viability before implementation.

\textbf{Types of Feasibility Analysis:}

\begin{longtable}[]{@{}lll@{}}
\toprule\noalign{}
\textbf{Analysis Type} & \textbf{Focus Area} & \textbf{Key Questions} \\
\midrule\noalign{}
\endhead
\bottomrule\noalign{}
\endlastfoot
\textbf{Technical} & Technology and processes & Can it be built? \\
\textbf{Economic} & Financial viability & Is it profitable? \\
\textbf{Market} & Demand and competition & Will it sell? \\
\textbf{Legal} & Regulatory compliance & Is it legal? \\
\textbf{Operational} & Implementation capability & Can we manage it? \\
\textbf{Schedule} & Time constraints & Can we deliver on time? \\
\end{longtable}

\textbf{Benefits:}

\begin{itemize}
\tightlist
\item
  \textbf{Risk Reduction}: Identifies potential problems early
\item
  \textbf{Resource Optimization}: Prevents wasteful investments
\item
  \textbf{Decision Support}: Provides data for go/no-go decisions
\item
  \textbf{Stakeholder Confidence}: Builds investor trust
\end{itemize}

\textbf{Process Flow:}

\begin{enumerate}
\tightlist
\item
  Define project scope and objectives
\item
  Conduct various feasibility analyses
\item
  Evaluate results and alternatives
\item
  Make implementation recommendation
\end{enumerate}

\end{solutionbox}
\begin{mnemonicbox}
``Technical Economic Market Legal Operational
Schedule reduces Risk, Optimizes Resources, Supports Decisions, builds
Confidence''

\end{mnemonicbox}

\subsection*{Question 5(a) OR [3
marks]}\label{q5a}

\textbf{What is bankruptcy? Write down the reasons of bankruptcy in
business.}

\begin{solutionbox}

\textbf{Bankruptcy Definition:} Legal process where individuals or
businesses unable to pay debts seek relief from creditors.

\textbf{Reasons for Business Bankruptcy:}

\begin{itemize}
\tightlist
\item
  \textbf{Poor Financial Management}: Inadequate cash flow control
\item
  \textbf{Market Competition}: Loss of market share
\item
  \textbf{Economic Recession}: External economic factors
\item
  \textbf{Over-expansion}: Rapid growth without proper planning
\item
  \textbf{High Debt Burden}: Excessive borrowing costs
\end{itemize}

\end{solutionbox}
\begin{mnemonicbox}
``Poor Market Economic Over High''

\end{mnemonicbox}

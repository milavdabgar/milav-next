\begin{solutionbox}

\begin{tabulary}{\linewidth}{|L|L|L|}
\hline

\questionmarks{3(c)}{7}{Explain Break-even analysis for a business practice.}

\begin{solutionbox}
\textbf{Break-even Formula:}
\[ \text{Break-even Point} = \frac{\text{Fixed Costs}}{\text{Selling Price per unit} - \text{Variable Cost per unit}} \]

\begin{center}
\begin{tikzpicture}[node distance=1.5cm, auto]
    \node [gtu block] (A) {Fixed Costs};
    \node [gtu block, right=of A] (B) {Variable Costs};
    \node [gtu block, below=of A] (C) {Total Costs};
    \node [gtu block, right=of C] (D) {Revenue Line};
    \node [gtu block, below=of C, xshift=2cm] (E) {Break-even Point};
    
    \path [gtu arrow] (A) -- (C);
    \path [gtu arrow] (B) -- (C);
    \path [gtu arrow] (C) -- (E);
    \path [gtu arrow] (D) -- (E);
\end{tikzpicture}
\captionof{figure}{Break-even Analysis Flow}
\end{center}

\textbf{Components:}

\begin{itemize}
    \item \keyword{Fixed Costs}: Rent, salaries, insurance
    \item \keyword{Variable Costs}: Raw materials, labor
    \item \keyword{Selling Price}: Revenue per unit
    \item \keyword{Contribution Margin}: Price minus variable cost
\end{itemize}

\textbf{Benefits:}

\begin{itemize}
    \item \keyword{Profit Planning}: Determines minimum sales needed
    \item \keyword{Price Setting}: Helps in pricing decisions
    \item \keyword{Cost Control}: Identifies cost structure
    \item \keyword{Investment Decisions}: Evaluates project viability
\end{itemize}
\end{solutionbox}

\begin{mnemonicbox}
\mnemonic{"Fixed Variable Selling Contribution makes Profit Price Cost Investment"}
\end{mnemonicbox}

\textbf{Bankruptcy Definition:} Legal process where individuals or businesses unable to pay debts seek relief from creditors.

\textbf{Reasons for Business Bankruptcy:}

\begin{itemize}
    \item \keyword{Poor Financial Management}: Inadequate cash flow control
    \item \keyword{Market Competition}: Loss of market share
    \item \keyword{Economic Recession}: External economic factors
    \item \keyword{Over-expansion}: Rapid growth without proper planning
    \item \keyword{High Debt Burden}: Excessive borrowing costs
\end{itemize}
\end{solutionbox}

\begin{mnemonicbox}
\mnemonic{"Poor Market Economic Over High"}
\end{mnemonicbox}

\questionmarks{5(b OR)}{4}{Distinguish between fixed cost and variable cost.}

\begin{solutionbox}
\begin{center}
\captionof{table}{Fixed vs Variable Cost}
\begin{tabulary}{\linewidth}{|L|L|L|}
\hline
\textbf{Aspect} & \textbf{Fixed Cost} & \textbf{Variable Cost} \\ \hline
\textbf{Nature} & Remains constant & Changes with production \\ \hline
\textbf{Examples} & Rent, insurance, salaries & Raw materials, labor \\ \hline
\textbf{Behavior} & Independent of output & Directly proportional \\ \hline
\textbf{Control} & Difficult to change & Easier to manage \\ \hline
\end{tabulary}
\end{center}
\end{solutionbox}

\begin{mnemonicbox}
\mnemonic{"Fixed Remains, Variable Changes"}
\end{mnemonicbox}

\questionmarks{5(c OR)}{7}{Describe various business ethics and their importance.}

\begin{solutionbox}
\textbf{Business Ethics Types:}

\begin{center}
\captionof{table}{Business Ethics Areas}
\begin{tabulary}{\linewidth}{|L|L|L|}
\hline
\textbf{Ethics Area} & \textbf{Description} & \textbf{Practices} \\ \hline
\textbf{Corporate Responsibility} & Company's duty to society & CSR activities, environmental care \\ \hline
\textbf{Employee Relations} & Fair treatment of workers & Equal opportunities, safety \\ \hline
\textbf{Customer Relations} & Honest dealing with clients & Quality products, fair pricing \\ \hline
\textbf{Supplier Relations} & Ethical procurement & Fair contracts, timely payments \\ \hline
\textbf{Environmental Ethics} & Sustainable practices & Waste reduction, green technology \\ \hline
\end{tabulary}
\end{center}

\textbf{Importance:}

\begin{itemize}
    \item \keyword{Reputation Building}: Creates positive brand image
    \item \keyword{Stakeholder Trust}: Builds confidence among all parties
    \item \keyword{Legal Compliance}: Avoids regulatory issues
    \item \keyword{Employee Motivation}: Improves workplace culture
    \item \keyword{Long-term Success}: Ensures sustainable growth
    \item \keyword{Competitive Advantage}: Differentiates from competitors
\end{itemize}

\textbf{Implementation Framework:}

\begin{enumerate}
    \item Develop ethical codes and policies
    \item Provide ethics training to employees
    \item Establish reporting mechanisms
    \item Monitor and evaluate ethical practices
    \item Take corrective actions when needed
\end{enumerate}

\textbf{Benefits for Business:}

\begin{itemize}
    \item \keyword{Risk Management}: Prevents ethical scandals
    \item \keyword{Customer Loyalty}: Builds lasting relationships
    \item \keyword{Investor Confidence}: Attracts ethical investors
    \item \keyword{Regulatory Support}: Maintains good government relations
\end{itemize}
\end{solutionbox}

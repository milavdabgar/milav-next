    \item \textbf{જોખમ મૂલ્યાંકન}: સંભવિત પડકારો અને તેના ઉકેલની ઓળખ કરે છે
\end{itemize}

\questionmarks{2(b)}{4}{બ્રેક-ઈવન પોઈન્ટ (in terms of sales revenue)ની ગણતરી કેઈ રીતે કરશો? તેનું graphical representation ઉદાહરણ સહિત આપો.}

\begin{solutionbox}
\textbf{સૂત્ર:}
\[ \text{બ્રેક-ઈવન પોઈન્ટ (વેચાણ)} = \frac{\text{સ્થિર ખર્ચ}}{\text{યોગદાન માર્જિન રેશિયો}} \]

જ્યાં: \[ \text{યોગદાન માર્જિન રેશિયો} = \frac{\text{વેચાણ} - \text{વેરિયેબલ કોસ્ટ}}{\text{વેચાણ}} \]

\textbf{ઉદાહરણ:}
\begin{itemize}
    \item સ્થિર ખર્ચ = રૂ. 50,000
    \item યુનિટ દીઠ વેચાણ કિંમત = રૂ. 100
    \item યુનિટ દીઠ વેરિયેબલ કોસ્ટ = રૂ. 60
    \item યુનિટ દીઠ યોગદાન = રૂ. 40
    \item યોગદાન માર્જિન રેશિયો = 40\%
    \item બ્રેક-ઈવન વેચાણ = $50,000 \div 0.40 = $ રૂ. 1,25,000
\end{itemize}

    \item \textbf{ઓનલાઇન સર્વે}: વ્યાપક પહોંચ માટે ડિજિટલ પ્રશ્નાવલી
\end{itemize}

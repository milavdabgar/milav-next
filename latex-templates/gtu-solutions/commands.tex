
% Custom commands for GTU solutions
% This file defines semantic commands for consistent formatting

% ============================================
% CONVERSION TO MARKDOWN
% ============================================
% To convert LaTeX solutions to Markdown for web (Hugo/Next.js):
%
% pandoc input.tex -o output.md --to=gfm --standalone --shift-heading-level-by=1
%
% This produces:
% - YAML frontmatter (title, author, date)
% - \section{} → H2 (Question 1)
% - \subsection{} → H3 (Question 1(a))
% - \subsubsection{} → H4 (Solution)
% - \paragraph{} → H5 (Given Data, Step 1, etc.)
% - Perfect GFM pipe tables
% - Blockquotes for mnemonics
%
% Your CMS controls the H1 from frontmatter title!
% ============================================

% Use standard LaTeX sectioning:
% \section{Question 1} → H2
% \subsection{Question 1(a) [3 marks] ...} → H3
% \subsubsection{Solution} → H4
% \paragraph{Details:} → H5

% Proper table environment with caption (Non-floating for boxes)
\newenvironment{answertable}[1]{%
  \begin{center}
  \captionof{table}{#1}
}{%
  \end{center}
}

% Proper figure environment for diagrams (Non-floating for boxes)
\newenvironment{answerdiagram}[1]{%
  \begin{center}
  \captionof{figure}{#1}
}{%
  \end{center}
}

% Semantic markup for key terms
\newcommand{\keyword}[1]{\textbf{#1}}
\newcommand{\code}[1]{\texttt{#1}}
\newcommand{\classname}[1]{\texttt{#1}}
\newcommand{\methodname}[1]{\texttt{#1}}

% Proper quotation marks for mnemonics with color
\newcommand{\mnemonic}[1]{\textcolor{mnemoniccolor}{``#1''}}
